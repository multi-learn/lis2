import numpy as np
import skimage.exposure as expo



[docs]def normalize_direct(x, xmin=None, zmax=None):
    """
    Normalize the data by directly projecting them on [0, 1]

    Parameters
    ----------
    x: np.ndarray
        The data

    Returns
    -------
    A normalized copy of the data
    """
    z = x.copy()
    if xmin is None:
        xmin = np.min(x.flat)
    z -= xmin
    if zmax is None:
        zmax = np.max(z.flat)
    if zmax > 0:
        z /= zmax
    return z




[docs]def normalize_histo(x):
    """
    Normalize the data with a histogram equalization

    Parameters
    ----------
    x: np.ndarray
        The data

    Returns
    -------
    A normalized copy of the data
    """
    z = normalize_direct(x)
    z = expo.equalize_hist(z)
    return z
